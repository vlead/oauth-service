* Dockerfile 
  To set up on a docker we have the following docker file.
#+BEGIN_SRC :eval no :tangle Dockerfile
FROM openjdk:8

COPY target/cas.war .

CMD java -jar cas.war

EXPOSE 8443

#+END_SRC
